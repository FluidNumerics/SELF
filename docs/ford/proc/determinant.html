  RECURSIVE FUNCTION Determinant(A,N) RESULT(D)

    IMPLICIT NONE
    INTEGER    :: N
    REAL(prec) :: A(1:N,1:N)
    REAL(prec) :: D
    ! LOCAL
    REAL(prec) :: M(1:N - 1,1:N - 1)
    INTEGER    :: j

    IF (N == 2) THEN
      D = A(1,1)*A(2,2) - A(1,2)*A(2,1)
      RETURN
    ELSE
      D = 0.0_prec
      DO j = 1,N
        M = GetMinor(A,1,j,N)
        D = D + (-1.0_prec)**(j + 1)*A(1,j)*Determinant(M,N - 1)
      END DO
    END IF

  END FUNCTION Determinant
