  FUNCTION Invert_2x2(A) RESULT(Ainv)

    IMPLICIT NONE
    REAL(prec) :: A(1:2,1:2)
    REAL(prec) :: Ainv(1:2,1:2)
    ! LOCAL
    REAL(prec) :: detA

    detA = Determinant(A,2)

    Ainv(1,1) = A(2,2)/detA
    Ainv(2,2) = A(1,1)/detA
    Ainv(1,2) = -A(1,2)/detA
    Ainv(2,1) = -A(2,1)/detA

  END FUNCTION Invert_2x2
