                          subroutine hipblasCheck(hipblasError_t)
                            use SELF_GPU_enums
                            use iso_c_binding
                            implicit none
                            integer(c_int) :: hipblasError_t

                            if(hipblasError_t /= HIPBLAS_STATUS_SUCCESS) then
                              write(*,*) "GPUBLAS ERROR: Error code = ",hipblasError_t
                              call exit(hipblasError_t)
                            endif
